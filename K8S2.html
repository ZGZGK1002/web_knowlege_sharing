Create a Deployment

写deployment的yml文件，管理启动哪几个image, 启动几个

apiVersion: extensions/v1beta1

kind: Deployment

metadata:

name: node

spec:

replicas: 1

template:

metadata:

labels:

app: node

spec:

containers:

- image: node:v1

name: mynode

imagePullPolicy: Never

ports:

- containerPort: 8080
